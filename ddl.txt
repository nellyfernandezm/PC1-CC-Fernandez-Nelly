CREATE TABLE IF NOT EXISTS pc_ml_cloud (
    id bigint generated by default as identity not null,
    mean_radius real,
    mean_texture real,
    mean_perimeter real,
    mean_area real,
    mean_smoothness real,
    prediction integer,
    probability real,
    created_time TIMESTAMP WITH TIME ZONE DEFAULT now(),
    constraint tb_ml_pkey primary key (id)
);
